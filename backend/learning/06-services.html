⚙️ Services Layer

Services are the brain. Every business rule, validation, role check, audit log call, and data orchestration lives here. Services are completely unaware of HTTP — they just receive values and return values.

The Golden Rule of Services

Services know nothing about HTTP. No req, no res, no status codes, no cookies. They receive plain JavaScript values (strings, numbers, objects) and return plain JavaScript values. If something goes wrong, they throw a ServiceError. The route layer is responsible for translating that into an HTTP response.

const projectRepo = require('../repositories/projectRepository');
const { recordStatusChange, recordFieldChange, setResponsibles, isWithinDeletionWindow }
  = require('../db/helpers');
const ServiceError = require('../utils/serviceError');

// ── list() ────────────────────────────────────────────────────
function list(user, { section_id, archived }) {
  const showArchived = archived === '1' ? 1 : 0;

  // Business rule: section_head is ALWAYS restricted to their own section.
  // Even if they pass ?section_id=1, we override it with their actual section.
  const sectionId = user.role === 'section_head'
    ? user.section_id
    : section_id;

  return projectRepo.enrich(projectRepo.findAll({ sectionId, showArchived }));
}

// ── create() ──────────────────────────────────────────────────
function create(user, body) {
  const { title, section_id, responsible_ids, status, future_plan,
          problems, row_order, progress, due_date } = body;

  // Validation: title is required
  if (!title?.trim()) throw new ServiceError(400, 'Title required');

  // section_head always creates in their own section
  const sectionId = user.role === 'section_head' ? user.section_id : section_id;
  if (!sectionId) throw new ServiceError(400, 'section_id required');

  const initialStatus = status || 'not_started';

  // Clamp progress to valid range 0-100
  // Math.min(100, Math.max(0, value)) ensures it can never be <0 or >100
  const initialProgress = Math.min(100, Math.max(0, Number(progress) || 0));

  const project = projectRepo.create({
    title: title.trim(), sectionId, status: initialStatus,
    future_plan: future_plan || '', problems: problems || '',
    row_order, progress: initialProgress, due_date,
  });

  // Audit trail: log the initial status (from null to 'not_started')
  recordStatusChange('project', project.id, null, initialStatus, user.id);

  // Set responsible people (many-to-many relationship)
  if (Array.isArray(responsible_ids)) {
    setResponsibles('project_responsibles', 'project_id', project.id, responsible_ids);
  }

  // Return enriched project (with responsibles and custom_values attached)
  return projectRepo.enrich([project])[0];
}

// ── update() ──────────────────────────────────────────────────
function update(user, id, body) {
  const project = projectRepo.findById(id);
  if (!project) throw new ServiceError(404, 'Not found');

  // section_head can only edit their own section's projects
  if (user.role === 'section_head' && project.section_id !== user.section_id) {
    throw new ServiceError(403, 'Forbidden');
  }

  const { title, responsible_ids, status, future_plan, problems,
          row_order, custom_values, progress, due_date } = body;

  // Nullish coalescing: use new value if provided, otherwise keep existing
  // Note: ?? preserves false, 0, and "" but replaces null/undefined
  const newStatus = status ?? project.status;
  const newProgress = progress != null
    ? Math.min(100, Math.max(0, Number(progress)))
    : (project.progress ?? 0);

  // undefined means "not sent" (keep existing); null means "clear the value"
  const newDueDate = due_date !== undefined
    ? (due_date || null)  // empty string → null
    : project.due_date;

  const updated = projectRepo.update(id, {
    title: title ?? project.title,
    status: newStatus, future_plan: future_plan ?? project.future_plan,
    problems: problems ?? project.problems, row_order: row_order ?? project.row_order,
    progress: newProgress, due_date: newDueDate,
  });

  // Audit trails (only record if something actually changed)
  recordStatusChange('project', id, project.status, newStatus, user.id);
  if (newDueDate !== project.due_date) {
    recordFieldChange('project', id, 'due_date', project.due_date, newDueDate, user.id);
  }
  if (Array.isArray(responsible_ids)) {
    setResponsibles('project_responsibles', 'project_id', id, responsible_ids);
  }
  if (Array.isArray(custom_values)) projectRepo.upsertCustomValues(id, custom_values);

  return projectRepo.enrich([updated])[0];
}

// ── archive() ─────────────────────────────────────────────────
function archive(user, id, archiveFlag) {
  const project = projectRepo.findById(id);
  if (!project) throw new ServiceError(404, 'Not found');
  if (user.role === 'section_head' && project.section_id !== user.section_id) {
    throw new ServiceError(403, 'Forbidden');
  }
  const archive = archiveFlag === false ? 0 : 1;
  projectRepo.setArchived(id, archive);
  return { ok: true, is_archived: archive };
}

// ── softDelete() ──────────────────────────────────────────────
function softDelete(user, id) {
  const project = projectRepo.findById(id);
  if (!project) throw new ServiceError(404, 'Not found');
  if (user.role === 'section_head' && project.section_id !== user.section_id) {
    throw new ServiceError(403, 'Forbidden');
  }
  // Business rule: can only delete within 10 minutes of creation
  if (!isWithinDeletionWindow(project.created_at)) {
    throw new ServiceError(409, 'older_than_10_min');
  }
  projectRepo.softDelete(id);
  return { ok: true };
}

// Simple pass-throughs (no business logic needed)
function getHistory(id) { return projectRepo.getHistory(id); }
function getDeadlineChanges() { return projectRepo.getDeadlineChanges(); }

module.exports = { list, create, update, archive, softDelete, getHistory, getDeadlineChanges };

Key Concepts Demonstrated

Nullish Coalescing (??) vs OR (||)

// || replaces falsy values: false, 0, "", null, undefined
title || project.title  // if title is "" (empty string), falls back to project.title ← BUG!

// ?? replaces ONLY null and undefined
title ?? project.title  // if title is "" (empty string), uses "" ← correct

The Validation Pattern

// Guard clauses: throw early, return late
// Each guard has one job: check one thing and throw if it fails
if (!title?.trim()) throw new ServiceError(400, 'Title required');
if (!sectionId) throw new ServiceError(400, 'section_id required');
if (!project) throw new ServiceError(404, 'Not found');
if (wrongRole) throw new ServiceError(403, 'Forbidden');

// Happy path comes last, after all guards pass
return projectRepo.create(...);

The Audit Trail Pattern

Every time a status or important field changes, a record is written to the status_history table. This lets you view the full history of any project.

// Records: "project 5 changed from 'in_progress' to 'completed' by user 3"
recordStatusChange('project', project.id, oldStatus, newStatus, user.id);

// Records: "project 5's due_date changed from '2024-01-01' to '2024-03-01' by user 3"
recordFieldChange('project', project.id, 'due_date', oldDate, newDate, user.id);

CodeMeaningWhen thrown

403ForbiddenValid user but wrong role/section
404Not FoundThe resource doesn't exist
409ConflictBusiness rule violated (e.g. too old to delete)

Services Are Testable Without HTTP

Because services have no HTTP dependencies, you can call them directly in tests:

// A test can just call the service function directly
const projectService = require('../src/services/projectService');

test('section_head cannot update another section project', () => {
  const user = { id: 1, role: 'section_head', section_id: 2 };
  expect(() => projectService.update(user, 5, { title: '...' }))
    .toThrow('Forbidden');
});
