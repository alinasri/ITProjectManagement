Scenario A: Login

POST /api/auth/login with body { "username": "ali", "password": "secret123" }

Ali is a section_head for section 2. He has a valid account and the right password.

A1 — Network

Browser sends HTTP request

POST /api/auth/login HTTP/1.1
Content-Type: application/json

{"username":"ali","password":"secret123"}

Node.js receives this on port 3000 and hands it to the Express app.

A4 — services/authService.js

Service: validate credentials

src/services/authService.js → login()
function login({ username, password }) {
  // Step 1: look up the user by username
  const user = userRepo.findByUsername('ali');
  // → { id: 5, username: 'ali', password_hash: '$2a$10$...', role: 'section_head', section_id: 2, is_active: 1 }

  // Step 2: compare provided password against stored hash
  bcrypt.compareSync('secret123', user.password_hash)  // → true ✓

  // Step 3: account is active check
  user.is_active === 1  // → true ✓

  // Step 4: create the JWT
  const token = jwt.sign(
    { id: 5, username: 'ali', role: 'section_head', section_id: 2 },
    JWT_SECRET,
    { expiresIn: '8h' }
  );
  // token = "eyJhbGciOiJIUzI1NiJ9.eyJpZCI6NX0.abc123..."

  // Step 5: return to route handler
  return { token, user: { id: 5, username: 'ali', role: 'section_head', section_id: 2 } };
}
↓

A7 — Network

Browser receives response

HTTP/1.1 200 OK

B6 — services/projectService.js

Service: apply business rules

src/services/projectService.js → list()
function list(user, { section_id, archived }) {
  const showArchived = archived === '1' ? 1 : 0;
  // archived is undefined (not in query string) → showArchived = 0

  // Business rule: section_head is forced to their own section
  const sectionId = user.role === 'section_head'
    ? user.section_id   // 2 — from the token, not from req.query!
    : section_id;
  // sectionId = 2

  // Get the raw projects, then enrich them
  return projectRepo.enrich(projectRepo.findAll({ sectionId: 2, showArchived: 0 }));
}

B8 — repositories/projectRepository.js

Repository: enrich() — 2 more queries (not N!)

src/repositories/projectRepository.js → enrich()
const ids = [1, 3];

// Query 2: get ALL responsibles for ALL projects in one shot
db.prepare('SELECT pr.project_id, per.id, per.name
  FROM project_responsibles pr JOIN personnel per ON per.id = pr.personnel_id
  WHERE pr.project_id IN (?,?)').all(1, 3)
// → [{ project_id:1, id:4, name:'Sara' }, { project_id:1, id:7, name:'Reza' }]

// Query 3: get ALL custom values for ALL projects in one shot
db.prepare('SELECT cv.project_id, cv.column_id, cv.value, cc.column_name
  FROM custom_values cv JOIN custom_columns cc ON cc.id = cv.column_id
  WHERE cv.project_id IN (?,?)').all(1, 3)
// → [{ project_id:1, column_id:1, value:'High', column_name:'Priority' }]

// Group by project_id using Map, then attach:
return [
  { id: 1, title: 'Network Upgrade', ..., responsibles: [{id:4,name:'Sara'},{id:7,name:'Reza'}], custom_values: [{column_id:1,value:'High'}] },
  { id: 3, title: 'Server Migration', ..., responsibles: [], custom_values: [] },
]

Summary: The Full Call Stack

GET /api/projects?section_id=2 │ ├─ app.js │ cors(), express.json(), cookieParser() │ route match → routes/projects.js │ ├─ middleware/auth.js: requireAuth │ jwt.verify(token) → req.user = { id:5, role:'section_head', section_id:2 } │ DB check: is account active? → yes │ next() │ ├─ utils/routeUtils.js: wrap(200, fn) │ try { res.status(200).json( fn(req) ) } │ ├─ services/projectService.js: list(user, query) │ sectionId = user.section_id = 2 (section_head override) │ return projectRepo.enrich(projectRepo.findAll({ sectionId:2, showArchived:0 })) │ ├─ repositories/projectRepository.js: findAll({ sectionId:2, showArchived:0 }) │ SQL: SELECT * FROM projects WHERE section_id=2 AND is_archived=0 AND is_deleted=0 │ → raw rows: [{ id:1, ... }, { id:3, ... }] │ ├─ repositories/projectRepository.js: enrich([row1, row2]) │ SQL: SELECT responsibles WHERE project_id IN (1,3) → 1 query │ SQL: SELECT custom_values WHERE project_id IN (1,3) → 1 query │ attach to each row │ → enriched: [{ id:1, responsibles:[...], custom_values:[...] }, ...] │ └─ Response: HTTP 200 + JSON array of enriched projects

What Would Happen If Ali Tried to See Section 1?

// Ali (section_head, section 2) requests: GET /api/projects?section_id=1

// In projectService.list():
const sectionId = user.role === 'section_head'
  ? user.section_id  // ← 2, ignoring the query param!
  : section_id;

// He gets section 2's projects regardless of what he passed in the URL.
// The business rule enforces the restriction.
// No error is thrown — it just silently overrides his request.

What Would Happen If the Token Expired?

// In requireAuth:
req.user = jwt.verify(expiredToken, JWT_SECRET);
// jwt.verify() throws: "TokenExpiredError: jwt expired"
// The catch block runs:
res.status(401).json({ error: 'Invalid token' });
// The request stops here. The service and repository never run.
