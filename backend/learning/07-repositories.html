🗄️ Repositories Layer

Repositories are the only layer that talks to the database. They translate between JavaScript values and SQL. They contain no business logic — just efficient data access.

What Is a Prepared Statement?

A prepared statement is SQL that is compiled once and then reused many times with different values. Think of it like a form with blank fields:

SELECT * FROM projects WHERE section_id = ? AND is_archived = ? ↑ ↑ placeholder placeholder At module load: compiled query plan is cached At call time: fill in: section_id = 2, is_archived = 0 → run query
// Prepared once when the module is first loaded:
const stmts = {
  findBySection: db.prepare('SELECT * FROM projects WHERE section_id = ? AND is_archived = ? AND is_deleted = 0'),
  findById: db.prepare('SELECT * FROM projects WHERE id = ?'),
  // ... more statements
};

// Used many times — each call just binds different values:
stmts.findBySection.all(2, 0);  // section 2, not archived
stmts.findBySection.all(3, 0);  // section 3, not archived

Two benefits of prepared statements: (1) Speed — the database parses the SQL only once. (2) Security — values are always treated as data, never as SQL, so SQL injection is impossible.

better-sqlite3 API: .get(), .all(), .run()

This project uses the better-sqlite3 library (synchronous SQLite). Three methods you'll see everywhere:

MethodReturnsUse for
.get(params)One row (or undefined)Find by ID, find one record
.all(params)Array of rowsList queries
.run(params){ changes, lastInsertRowid }INSERT, UPDATE, DELETE
stmts.findById.get(5)         // → { id: 5, title: '...', ... } or undefined
stmts.findAll.all(0)          // → [{ id:1, ... }, { id:2, ... }, ...]
stmts.create.run('title', ...) // → { changes: 1, lastInsertRowid: 6 }

The Full projectRepository.js

File: src/repositories/projectRepository.js

const db = require('../db/schema');  // single shared DB connection

// All SQL compiled once at module load
const stmts = {
  findBySection: db.prepare('SELECT * FROM projects WHERE section_id = ? AND is_archived = ? AND is_deleted = 0 ORDER BY row_order, id'),
  findAll:       db.prepare('SELECT * FROM projects WHERE is_archived = ? AND is_deleted = 0 ORDER BY section_id, row_order, id'),
  findById:      db.prepare('SELECT * FROM projects WHERE id = ?'),
  maxOrder:      db.prepare('SELECT COALESCE(MAX(row_order),0) as m FROM projects WHERE section_id = ?'),

  create: db.prepare(`INSERT INTO projects
    (title, section_id, status, future_plan, problems, row_order, progress, due_date)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?)`),

  update: db.prepare(`UPDATE projects
    SET title = ?, status = ?, future_plan = ?, problems = ?,
        row_order = ?, progress = ?, due_date = ?, updated_at = datetime('now')
    WHERE id = ?`),

  setArchived: db.prepare("UPDATE projects SET is_archived = ?, updated_at = datetime('now') WHERE id = ?"),
  softDelete:  db.prepare('UPDATE projects SET is_deleted = 1 WHERE id = ?'),

  upsertCustom: db.prepare(`INSERT INTO custom_values (project_id, column_id, value) VALUES (?, ?, ?)
    ON CONFLICT(project_id, column_id) DO UPDATE SET value = excluded.value`),

  getHistory: db.prepare(`SELECT sh.*, u.username as changed_by_username
    FROM status_history sh LEFT JOIN users u ON u.id = sh.changed_by
    WHERE sh.entity_type = 'project' AND sh.entity_id = ?
    ORDER BY sh.changed_at DESC`),
};

// ── enrich() ──────────────────────────────────────────────────
// Attaches responsibles and custom_values to plain project rows.
// The KEY function — solves the N+1 problem.
function enrich(projects) {
  if (projects.length === 0) return [];

  const ids = projects.map(p => p.id);
  const placeholders = ids.map(() => '?').join(',');  // "?,?,?" for 3 projects

  // ONE query for ALL responsibles across ALL projects
  const allResponsibles = db.prepare(
    `SELECT pr.project_id, per.id, per.name
     FROM project_responsibles pr JOIN personnel per ON per.id = pr.personnel_id
     WHERE pr.project_id IN (${placeholders}) ORDER BY per.name`
  ).all(...ids);

  // ONE query for ALL custom values across ALL projects
  const allCustomValues = db.prepare(
    `SELECT cv.project_id, cv.column_id, cv.value, cc.column_name
     FROM custom_values cv JOIN custom_columns cc ON cc.id = cv.column_id
     WHERE cv.project_id IN (${placeholders})`
  ).all(...ids);

  // Group results into Maps for O(1) lookup
  // Map: { project_id → [responsible1, responsible2, ...] }
  const responsiblesMap = new Map(ids.map(id => [id, []]));
  for (const r of allResponsibles) {
    responsiblesMap.get(r.project_id).push({ id: r.id, name: r.name });
  }

  const customValuesMap = new Map(ids.map(id => [id, []]));
  for (const cv of allCustomValues) {
    customValuesMap.get(cv.project_id).push({ column_id: cv.column_id, value: cv.value, column_name: cv.column_name });
  }

  // Spread the project + attach the grouped arrays
  return projects.map(p => ({
    ...p,
    responsibles: responsiblesMap.get(p.id),
    custom_values: customValuesMap.get(p.id),
  }));
}

function findAll({ sectionId, showArchived }) {
  if (sectionId) return stmts.findBySection.all(sectionId, showArchived);
  return stmts.findAll.all(showArchived);
}

function findById(id) { return stmts.findById.get(id); }

function create({ title, sectionId, status, future_plan, problems, row_order, progress, due_date }) {
  const maxOrder = stmts.maxOrder.get(sectionId).m;
  const result = stmts.create.run(
    title, sectionId, status, future_plan, problems,
    row_order ?? maxOrder + 1,  // auto-place at end if no row_order given
    progress, due_date ?? null
  );
  return findById(result.lastInsertRowid);  // return the new row
}

function update(id, { title, status, future_plan, problems, row_order, progress, due_date }) {
  stmts.update.run(title, status, future_plan, problems, row_order, progress, due_date, id);
  return findById(id);  // re-fetch to get updated_at timestamp
}

function setArchived(id, archive) { stmts.setArchived.run(archive, id); }
function softDelete(id) { stmts.softDelete.run(id); }

function upsertCustomValues(projectId, customValues) {
  customValues.forEach(({ column_id, value }) =>
    stmts.upsertCustom.run(projectId, column_id, value ?? '')
  );
}

function getHistory(id) { return stmts.getHistory.all(id); }

module.exports = { enrich, findAll, findById, create, update, setArchived, softDelete, upsertCustomValues, getHistory };

The N+1 Query Problem

This is one of the most common performance mistakes. Imagine you have 50 projects and you want each project with its list of responsible people.

N+1 Problem: 1 query to get all projects + 1 query PER PROJECT to get its responsibles = 51 queries for 50 projects. As you scale, this becomes 1001 queries for 1000 projects.

❌ N+1 approach (bad): SELECT * FROM projects WHERE section_id = 2 ← 1 query (returns 50 rows) SELECT * FROM project_responsibles WHERE id = 1 ← 1 query per project SELECT * FROM project_responsibles WHERE id = 2 SELECT * FROM project_responsibles WHERE id = 3 ...50 times... Total: 51 queries ✅ Bulk approach (used here): SELECT * FROM projects WHERE section_id = 2 ← 1 query SELECT * FROM project_responsibles WHERE id IN (1,2,3,...50) ← 1 query for ALL Total: 2 queries

The enrich() function uses the bulk approach, then organizes results into a Map for O(1) lookup when attaching responsibles to each project.

Soft Delete vs Hard Delete

Instead of actually removing rows from the database, this app marks them as deleted:

// Hard delete — data is gone forever
DELETE FROM projects WHERE id = 5;

// Soft delete — data is preserved, just hidden
UPDATE projects SET is_deleted = 1 WHERE id = 5;

// All queries filter out soft-deleted rows:
SELECT * FROM projects WHERE is_deleted = 0 ...
// The deleted project still exists in the DB, just invisible to queries

Why? Soft delete lets you:

  • Preserve the audit trail (status_history still references the project)
  • Potentially "undelete" if something was deleted by mistake
  • Maintain referential integrity (other records still point to the row)

The Upsert Pattern

An upsert is "insert if not exists, update if it does exist". SQLite's ON CONFLICT ... DO UPDATE does this in one statement:

INSERT INTO custom_values (project_id, column_id, value) VALUES (?, ?, ?)
ON CONFLICT(project_id, column_id)
DO UPDATE SET value = excluded.value;

// If (project_id=5, column_id=2) doesn't exist → INSERT
// If (project_id=5, column_id=2) already exists → UPDATE value
// "excluded" refers to the row that would have been inserted

COALESCE in SQL

SELECT COALESCE(MAX(row_order), 0) as m FROM projects WHERE section_id = ?
// MAX(row_order) returns NULL if there are no rows
// COALESCE(NULL, 0) returns 0 — safe default

Repositories should have no if statements about business rules. If you see something like "if user is section_head, filter by section" in a repository, it belongs in the service layer instead.
