1. ServiceError — Custom Error With HTTP Status

File: src/utils/serviceError.js

// Extends the built-in Error class to add an HTTP status code
class ServiceError extends Error {
  constructor(status, message) {
    super(message);    // sets this.message
    this.status = status;  // adds this.status (e.g. 404)
  }
}

module.exports = ServiceError;

Without ServiceError, you couldn't tell a route handler "respond with 404". Plain JavaScript Error has no concept of HTTP. By extending it with a status property, the wrap() function can read it and send the right HTTP code:

// In a service:
throw new ServiceError(404, 'Project not found');
// err.status = 404, err.message = 'Project not found'

// In wrap():
res.status(err.status || 500).json({ error: err.message });
// sends: HTTP 404 { "error": "Project not found" }

What happens with a regular Error?

throw new Error('Something broke');
// err.status is undefined
// wrap(): err.status || 500 → 500
// sends: HTTP 500 { "error": "Something broke" }

2. wrap() — The Route Error Handler

File: src/utils/routeUtils.js

function wrap(statusCode, fn) {
  // Returns Express middleware: (req, res) => { ... }
  return (req, res) => {
    try {
      // fn(req) calls the service function
      // The return value (data) is sent as JSON with the success status code
      res.status(statusCode).json(fn(req));
    } catch (err) {
      // ServiceError has .status; plain Error doesn't (falls back to 500)
      res.status(err.status || 500).json({ error: err.message });
    }
  };
}

module.exports = { wrap };

Notice that wrap() returns a function — it doesn't run immediately. Express stores that returned function and calls it later when a request arrives. This is the same "higher-order function" pattern as requireRole().

The Login Route Exception

The login route can't use wrap() because it needs to set a cookie before sending the response. Only the login handler needs this special treatment:

// Login can't use wrap() — needs to set a cookie
router.post('/login', (req, res) => {
  try {
    const { token, user } = authService.login(req.body);
    res.cookie('token', token, { httpOnly: true, sameSite: 'lax', maxAge: 8 * 60 * 60 * 1000 });
    res.json({ token, user });
  } catch (err) {
    res.status(err.status || 500).json({ error: err.message });
  }
});

// Everything else uses wrap()
router.get('/me', requireAuth, wrap(200, req => authService.getProfile(req.user.id)));

3. db/helpers.js — Shared Database Operations

File: src/db/helpers.js

These are operations that would otherwise be duplicated across multiple service files. Every service that needs an audit trail or has responsible people calls these helpers.

const db = require('./schema');

// ── recordStatusChange() ───────────────────────────────────────
// Called every time an entity's status changes.
// Skips recording if the status didn't actually change.
function recordStatusChange(entityType, entityId, fromStatus, toStatus, userId) {
  if (fromStatus === toStatus) return;  // no-op
  try {
    db.prepare(`INSERT INTO status_history
      (entity_type, entity_id, from_status, to_status, changed_by)
      VALUES (?, ?, ?, ?, ?)`)
      .run(entityType, entityId, fromStatus ?? null, toStatus, userId ?? null);
  } catch (err) {
    console.error('[status_history] insert failed:', err.message);
    // Non-fatal: a failed audit log should not break the main operation
  }
}

// ── recordFieldChange() ───────────────────────────────────────
// Like recordStatusChange but for specific fields (e.g., due_date).
function recordFieldChange(entityType, entityId, field, fromValue, toValue, userId) {
  if (fromValue === toValue) return;
  try {
    db.prepare(`INSERT INTO status_history
      (entity_type, entity_id, field, from_status, to_status, changed_by)
      VALUES (?, ?, ?, ?, ?, ?)`)
      .run(entityType, entityId, field, fromValue ?? null, toValue ?? '', userId ?? null);
  } catch (err) {
    console.error('[status_history] field change failed:', err.message);
  }
}

// ── setResponsibles() ─────────────────────────────────────────
// Replace the entire list of responsible people for an entity.
// Strategy: DELETE all existing → INSERT the new ones.
// Called for projects, ongoing-tasks, etc.
function setResponsibles(table, fk, id, personnelIds) {
  db.prepare(`DELETE FROM ${table} WHERE ${fk} = ?`).run(id);
  const insert = db.prepare(`INSERT INTO ${table} (${fk}, personnel_id) VALUES (?, ?)`);
  personnelIds.forEach(pid => insert.run(id, pid));
}

// ── setSections() ─────────────────────────────────────────────
// Replace the list of sections for an entity (purchases, tenders, contracts).
function setSections(table, fkCol, id, sectionIds) {
  db.prepare(`DELETE FROM ${table} WHERE ${fkCol} = ?`).run(id);
  const insert = db.prepare(`INSERT INTO ${table} (${fkCol}, section_id) VALUES (?, ?)`);
  sectionIds.forEach(sid => insert.run(id, sid));
}

// ── isWithinDeletionWindow() ──────────────────────────────────
// Returns true if the entity was created less than 10 minutes ago.
// Used to enforce the "10-minute delete window" business rule.
function isWithinDeletionWindow(createdAt) {
  // SQLite stores timestamps without timezone. We append 'Z' to tell
  // JavaScript to treat it as UTC, not local time.
  const ageMs = Date.now() - new Date(createdAt + 'Z').getTime();
  return ageMs <= 10 * 60 * 1000;  // 10 minutes in milliseconds
}

module.exports = { recordStatusChange, recordFieldChange, setResponsibles, setSections, isWithinDeletionWindow };

How the Audit Trail Works

Every status change and important field change is written to the status_history table. Here's what the records look like:

The GET /api/projects/:id/history endpoint reads this table and returns the full timeline for any project.

The setResponsibles Pattern

When you update who is responsible for a project, you could try to add/remove individual people. But it's simpler to just wipe the list and rebuild it:

// Before: project 5 is assigned to personnel [2, 4]
// After the update, we want personnel [2, 6]

// Delete approach (calculate diff — complex):
DELETE FROM project_responsibles WHERE project_id = 5 AND personnel_id = 4;
INSERT INTO project_responsibles (project_id, personnel_id) VALUES (5, 6);

// Replace-all approach (simple):
DELETE FROM project_responsibles WHERE project_id = 5;  ← wipe
INSERT INTO project_responsibles (project_id, personnel_id) VALUES (5, 2);  ← rebuild
INSERT INTO project_responsibles (project_id, personnel_id) VALUES (5, 6);

The replace-all approach is simpler and has the same end result. For small lists of people, the performance difference is negligible.

4. db/schema.js — Database Connection

File: src/db/schema.js

This file creates the SQLite connection and initializes the schema. Every repository imports it to get the database object:

const Database = require('better-sqlite3');
const path = require('path');

// Open (or create) the database file
const db = new Database(path.join(__dirname, '../../data/it_pm.db'));

// Enable Write-Ahead Logging — better performance for concurrent reads
db.pragma('journal_mode = WAL');

// Enforce foreign key constraints (SQLite disables them by default!)
db.pragma('foreign_keys = ON');

// Create all tables if they don't exist
// ... (many CREATE TABLE IF NOT EXISTS statements)

// Export the single connection object
module.exports = db;

Because Node.js caches modules, every file that does require('../db/schema') receives the same database connection object. You don't need to worry about managing multiple connections.
